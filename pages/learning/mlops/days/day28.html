1 Service Logic (CRUD)

func GetByID(id int) (Student, error) {
    for _, s := range students {
        if s.ID == id { return s, nil }
    }
    return Student{}, errors.New("not found")
}

func UpdateGPA(id int, gpa float64) error {
    for i := range students {
        if students[i].ID == id { students[i].GPA = gpa; return nil }
    }
    return errors.New("not found")
}
