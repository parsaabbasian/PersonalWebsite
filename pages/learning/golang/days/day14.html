student/student.go
package student

import "errors"

type Student struct {
    ID int
    Name string
    GPA float64
}

var students []Student

func UpdateGPA(id int, gpa float64) error {
    for i := range students {
        if students[i].ID == id {
            students[i].GPA = gpa
            return nil
        }
    }
    return errors.New("student not found")
}
main.go
package main

import (
    "fmt"
    "student-system/student"
)

func main() {
    student.Add(1, "Parsa", 7.75)
    err := student.UpdateGPA(1, 8.5)
    if err != nil {
        fmt.Println(err)
    }
}
